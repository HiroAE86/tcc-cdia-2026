% ---------------------------------------------------------------- 10. Alerta
\begin{frame}{Três sinais de alerta}
  \begin{enumerate}
    \item \textbf{Matriz de confusão assimétrica:} \textit{precision}(Desce) = 1{,}00 com \textit{recall}(Desce) = 0{,}20 --- previu ``Desce'' corretamente em só \alert{11 das 54 quedas reais}
    \item \textbf{Acurácia suspeita:} 76{,}3\% supera a classe majoritária da janela de teste (69{,}5\%; \textit{base rate} de treino 59/41) por só $\sim$7 pontos --- um ``sempre Sobe'' já acerta 69{,}5\%
    \item \textbf{Protocolo frágil:} estimativa pontual --- \alert{sem} intervalo de confiança, \alert{uma} semente, \alert{uma} janela temporal
  \end{enumerate}
  \vspace{1ex}
  \begin{block}{Decisão}
    Antes de aceitar o resultado: submetê-lo a protocolos progressivamente
    mais rigorosos. Se for real, sobrevive.
  \end{block}
  \vspace{0.5ex}
  {\footnotesize Treino 59/41; janela de teste 69{,}5\% (123/177) --- dois subconjuntos diferentes do \textit{split} cronológico, períodos distintos.}
\end{frame}

% ---------------------------------------------------------------- 11. Investigação
\begin{frame}{Bateria de investigação: 1.435 execuções --- pergunta e métrica definidas a priori}
  \begin{columns}[T]
    \column{0.55\textwidth}
    \begin{table}
      \footnotesize
      \begin{tabular}{lr}
        \toprule
        \textbf{Experimento} & \textbf{Runs} \\
        \midrule
        Baseline autoregressivo               & 4 \\
        Bootstrap CI                          & 1 \\
        Multi-seed (ITUB4)                    & 40 \\
        Multi-seed $\times$ multi-ticker      & 120 \\
        Ensemble + backtest                   & 20 \\
        Expanding-window CV                   & 145 \\
        VALE3 \textit{deep-dive}              & 880 \\
        Ablation PRICE/SENT/PRICE+SENT        & 225 \\
        \midrule
        \textbf{Total} & \textbf{1.435} \\
        \bottomrule
      \end{tabular}
    \end{table}
    \column{0.42\textwidth}
    \textbf{Protocolo (idêntico em todos)}
    \begin{itemize}
      \item Hiperparâmetros congelados; \texttt{StandardScaler} só no treino; bootstrap CI 1.000; Wilcoxon pareado
      \item Cada experimento = 1 pergunta e métrica definidas a priori; nunca escolhidas \textit{a posteriori}
      \item \alert{Oposto de \textit{p-hacking}:} protocolo idêntico \alert{destrói} o positivo, não seleciona o melhor
      \item Bonferroni (6 células) \alert{reforça o nulo} do sentimento \alert{e} o positivo ``preço vence'' ($|\Delta|=0{,}255\approx6\times$ std passa com folga)
    \end{itemize}
  \end{columns}
  \vspace{1ex}
  {\footnotesize Bonferroni / espaço de busca e reconciliação dos $\sim$80 \textit{sanity checks}: detalhe no Q\&A.}
\end{frame}

  \end{table}
  \vspace{1ex}

    \item XGBoost com só \texttt{return}, \texttt{lag\_1}, \texttt{lag\_5}, \texttt{Volume}, \texttt{std21} já atinge \textbf{0{,}658}
    \item Nesta janela única o IC é largo demais para separar 0{,}709 do \textit{baseline} --- a separação \textbf{só} emerge no CV (0{,}667 vs 0{,}509, slide 14). Janela única não decide nada; é exatamente o ponto do trabalho

  \end{itemize}
  \vspace{0.5ex}
  {\footnotesize \alert{0{,}658 $=$ 1 janela (IC largo); 0{,}667 $=$ média 5 \textit{folds} CV. \textbf{O preço vence o sentimento nos dois.}}}
\end{frame}

    {\scriptsize IC do $\Delta$ MÉDIO é estreito porque agrega 225 \textit{runs}; MDE 0{,}13--0{,}22 é \alert{por ativo} ($n$ pequeno). Não há contradição: o MESMO desenho detecta preço (0{,}667 estável). O nulo TEM IC; o 0{,}709 nunca teve. (Bonferroni / espaço de busca no Q\&A.)}
  \end{columns}
\end{frame}

% ---------------------------------------------------------------- 17. Síntese
\begin{frame}{Síntese da investigação}
  \begin{block}{Conclusão unificada}
    Sob avaliação correta (multi-\textit{fold}, multi-semente, IC, \textit{ablation}),
    as \textit{features} de sentimento FinBERT-PT-BR \alert{não adicionam sinal
    preditivo mensurável} ao \textit{baseline} autoregressivo --- em nenhum
    dos três ativos.
  \end{block}
  \vspace{1ex}
  \textbf{O AUC de 0{,}709 era um artefato de janela única, produzido por:}
  \begin{enumerate}
    \item Variância de semente (std 0{,}261 --- colapso bimodal do Transformer)
    \item Viés de amostragem da janela de teste específica
    \item Ausência de intervalos de confiança que revelariam a incerteza
  \end{enumerate}
  \vspace{1ex}
  \footnotesize Escopo: conclusão vale para a representação adotada (5 \textit{features} diárias FinBERT-PT-BR),
  não para toda forma de incorporar texto. Resultado é ``nenhum efeito detectável neste tamanho amostral''.
\end{frame}

\end{frame}

% ---------------------------------------------------------------- Backup: por que AUC + backtest
\begin{frame}{Reserva --- Por que AUC? E o \textit{backtest}}
  \begin{block}{Por que AUC é a métrica}
    {\footnotesize AUC mede \textbf{separabilidade direcional} (capacidade de ordenar ``sobe'' acima de ``desce'') --- exatamente a pergunta --- e era a métrica do 0{,}709 que eu auditava (consistência). O nulo é robusto: o mesmo desenho move a AUC no \textbf{preço} (0{,}667) e \alert{não} no sentimento. \textbf{Honestidade:} a \textit{ablation} (ligar/desligar o sentimento mantendo o resto) mede só $\Delta$AUC; calibração (Brier/ECE) foi computada na fase diagnóstica, não nos 225 \textit{runs}. Claim restrito: ``nenhum \textbf{ganho discriminativo} (de ordenação) detectável''.}
  \end{block}
  \begin{block}{O \textit{backtest} existe --- e confirma o nulo}
    {\footnotesize Com custo de 10 \textit{bps}: \textit{ensemble} Sharpe \alert{$-$1{,}29} ($-$0{,}10\%) \textit{vs} \textit{buy-and-hold} Sharpe 3{,}25 ($+$50{,}30\%). Fora do \textit{slide} principal por fragilidade (filtro \textit{val}-AUC$\geq$0{,}5 deixou 2/20 \textit{seeds}, anticorrelacionadas) --- \textbf{não} por ser desfavorável. Não escondi um positivo; tenho um negativo que \alert{reforça} a tese. \textbf{AUC $\neq$ retorno} nas duas direções.}
  \end{block}
\end{frame}
